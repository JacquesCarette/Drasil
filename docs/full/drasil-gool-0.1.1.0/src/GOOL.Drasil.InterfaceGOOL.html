  type Class r
  -- | Main external method for creating a class.
  --   Inputs: parent class, variables, constructor(s), methods

    [SMethod r] -> SClass r

    [SMethod r] -> SClass r

    [SMethod r] -> SClass r

  docClass :: String -> SClass r -> SClass r
